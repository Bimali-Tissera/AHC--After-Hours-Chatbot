**1. `src/app/admin/login/actions.ts`** — Login server action:

```typescript
'use server'
import { redirect } from 'next/navigation'
import { createClient } from '@/lib/supabase/server'

export async function login(formData: FormData) {
  const email = String(formData.get('email') ?? '')
  const password = String(formData.get('password') ?? '')

  const supabase = await createClient()
  const { error } = await supabase.auth.signInWithPassword({ email, password })

  if (error) {
    redirect('/admin/login?error=invalid_credentials')
  }

  redirect('/admin/leads')
}
```

**2. `src/app/admin/logout/actions.ts`** — Logout server action:

```typescript
'use server'
import { redirect } from 'next/navigation'
import { revalidatePath } from 'next/cache'

export async function logout() {
  const supabase = await createClient()
  await supabase.auth.signOut()
  revalidatePath('/', 'layout')
  redirect('/admin/login')
}
```

**1. `src/app/admin/layout.tsx`** — Auth guard + sidebar + main area shell:

This layout wraps ALL /admin/* pages EXCEPT /admin/login (login page is outside this layout's scope because the file structure places login as a sibling route — BUT in Next.js App Router, /admin/login IS nested under /admin/layout.tsx. So the layout must handle both authenticated and unauthenticated states).

The layout must:
1. Call `getUser()` (NOT getClaims, NOT getSession) for server-verified session liveness
2. If the current path is `/admin/login`, render children WITHOUT the sidebar (login page gets its own full-screen layout)
3. If no user AND path is not `/admin/login`, redirect to `/admin/login`
4. If user exists AND path is not `/admin/login`, render the sidebar + main area shell

To detect the current path, import `headers` from `next/headers` and read `x-pathname` — OR simpler: just always render children. The login page has its own full-screen layout, so the admin layout can just be: if no user and not login, redirect; otherwise render sidebar + children. But the login page would then show inside the sidebar layout when the user is already authenticated (before proxy redirects).

SIMPLEST CORRECT APPROACH: The admin layout checks auth. If no user, redirect to login. The login page is at `/admin/login` which means it's nested under this layout. To avoid redirect loops, the layout must NOT redirect if the current route IS the login page. Use the `headers()` API to read `x-next-pathname` or check `x-url`:

```typescript
import { redirect } from 'next/navigation'
import { headers } from 'next/headers'
import { createClient } from '@/lib/supabase/server'
import { logout } from '@/app/admin/logout/actions'

export default async function AdminLayout({ children }: { children: React.ReactNode }) {
  const supabase = await createClient()
  const { data: { user } } = await supabase.auth.getUser()

  const headersList = await headers()
  const pathname = headersList.get('x-next-pathname') || ''
  const isLoginPage = pathname === '/admin/login'

  if (!user && !isLoginPage) {
    redirect('/admin/login')
  }

  // Login page renders without the shell
  if (isLoginPage) {
    return <>{children}</>
  }

  // Authenticated admin shell
  return (
    <div className="min-h-screen bg-gray-50 flex">
      {/* Sidebar */}
      <aside className="bg-white border-r border-gray-200 w-64 flex-shrink-0 p-4 flex flex-col">
        <p className="text-sm font-semibold text-gray-900 mb-6">Ekwa Dental</p>
        <nav className="flex-1">
          <a
            href="/admin/leads"
            className="flex items-center text-sm text-blue-600 font-semibold bg-blue-50 border-l-2 border-blue-600 px-3 py-2 rounded-r"
          >
            Leads
          </a>
        </nav>
        <form>
          <button
            formAction={logout}
            className="text-sm text-gray-500 hover:text-gray-900 transition-colors"
          >
            Log out
          </button>
        </form>
      </aside>

      {/* Main content */}
      <main className="flex-1 p-8">
        {children}
      </main>
    </div>
  )
}
```

NOTE: If `x-next-pathname` header is not available in Next.js 16, use the `next/headers` + URL approach or restructure routes so login is outside the /admin layout. Test during implementation and adapt.
